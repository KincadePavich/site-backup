43 /* info about EdDSA curve specifically ed25519, defined as an elliptic curve
44  over GF(p) */
45 /*
46  32, key size
47  "ED25519", curve name
48  "2^255-19", prime number
49  "SHA512", hash function
50  "-121665/121666", value of d
51 */
52 
53 #define ED25519_KEY_SIZE 32 /* private key only */
54 #define ED25519_SIG_SIZE 64
55 
56 #define ED25519_PUB_KEY_SIZE 32 /* compressed */
57 /* both private and public key */
58 #define ED25519_PRV_KEY_SIZE (ED25519_PUB_KEY_SIZE+ED25519_KEY_SIZE)
59

66 /* An ED25519 Key */
67 struct ed25519_key {
68  byte p[ED25519_PUB_KEY_SIZE]; /* compressed public key */
69  byte k[ED25519_PRV_KEY_SIZE]; /* private key : 32 secret -- 32 public */
70 #ifdef FREESCALE_LTC_ECC
71  /* uncompressed point coordinates */
72  byte pointX[ED25519_KEY_SIZE]; /* recovered X coordinate */
73  byte pointY[ED25519_KEY_SIZE]; /* Y coordinate is the public key with The most significant bit of the final octet always zero. */
74 #endif
75 #ifdef WOLFSSL_ASYNC_CRYPT
76  WC_ASYNC_DEV asyncDev;
77 #endif
78 };
