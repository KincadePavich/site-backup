23 /* The memory tracker overrides the wolfSSL memory callback system and uses a
24  * static to track the total, peak and currently allocated bytes.
25  *
26  * If you are already using the memory callbacks then enabling this will
27  * override the memory callbacks and prevent your memory callbacks from
28  * working. This assumes malloc() and free() are available. Feel free to
29  * customize this for your needs.
30 
31  * The enable this feature define the following:
32  * #define USE_WOLFSSL_MEMORY
33  * #define WOLFSSL_TRACK_MEMORY
34  *
35  * On startup call:
36  * InitMemoryTracker();
37  *
38  * When ready to dump the memory report call:
39  * ShowMemoryTracker();
40  *
41  * Report example:
42  * total Allocs = 228
43  * total Bytes = 93442
44  * peak Bytes = 8840
45  * current Bytes = 0
46  *
47  *
48  * You can also:
49  * #define WOLFSSL_DEBUG_MEMORY
50  *
51  * To print every alloc/free along with the function and line number.
52  * Example output:
53  * Alloc: 0x7fa14a500010 -> 120 at wc_InitRng:496
54  * Free: 0x7fa14a500010 -> 120 at wc_FreeRng:606
55  */

108 #ifdef WOLFSSL_DEBUG_MEMORY
109  STATIC INLINE void* TrackMalloc(size_t sz, const char* func, unsigned int line)
110 #else
111  STATIC INLINE void* TrackMalloc(size_t sz)
112 #endif
113  {
114  memoryTrack* mt;
115 
116  if (sz == 0)
117  return NULL;
118 
119  mt = (memoryTrack*)malloc(sizeof(memoryTrack) + sz);
120  if (mt == NULL)
121  return NULL;
122 
123  mt->u.hint.thisSize = sz;
124  mt->u.hint.thisMemory = (byte*)mt + sizeof(memoryTrack);
125 
126 #ifdef WOLFSSL_DEBUG_MEMORY
127  printf("Alloc: %p -> %u at %s:%d\n", mt->u.hint.thisMemory, (word32)sz, func, line);
128 #endif
129 
130 #ifdef DO_MEM_STATS
131  ourMemStats.totalAllocs++;
132  ourMemStats.totalBytes += sz;
133  ourMemStats.currentBytes += sz;
134  if (ourMemStats.currentBytes > ourMemStats.peakBytes)
135  ourMemStats.peakBytes = ourMemStats.currentBytes;
136 #endif
137 
138  return mt->u.hint.thisMemory;
139  }

142 #ifdef WOLFSSL_DEBUG_MEMORY
143  STATIC INLINE void TrackFree(void* ptr, const char* func, unsigned int line)
144 #else
145  STATIC INLINE void TrackFree(void* ptr)
146 #endif
147  {
148  memoryTrack* mt;
149 
150  if (ptr == NULL) {
151  return;
152  }
153 
154  mt = (memoryTrack*)ptr;
155  --mt; /* same as minus sizeof(memoryTrack), removes header */
156 
157 #ifdef DO_MEM_STATS
158  ourMemStats.currentBytes -= mt->u.hint.thisSize;
159  ourMemStats.totalDeallocs++;
160 #endif
161 
162 #ifdef WOLFSSL_DEBUG_MEMORY
163  printf("Free: %p -> %u at %s:%d\n", ptr, (word32)mt->u.hint.thisSize, func, line);
164 #endif
165 
166  free(mt);
167  }

170 #ifdef WOLFSSL_DEBUG_MEMORY
171  STATIC INLINE void* TrackRealloc(void* ptr, size_t sz, const char* func, unsigned int line)
172 #else
173  STATIC INLINE void* TrackRealloc(void* ptr, size_t sz)
174 #endif
175  {
176  #ifdef WOLFSSL_DEBUG_MEMORY
177  void* ret = TrackMalloc(sz, func, line);
178  #else
179  void* ret = TrackMalloc(sz);
180  #endif
181 
182  if (ptr) {
183  /* if realloc is bigger, don't overread old ptr */
184  memoryTrack* mt = (memoryTrack*)ptr;
185  --mt; /* same as minus sizeof(memoryTrack), removes header */
186 
187  if (mt->u.hint.thisSize < sz)
188  sz = mt->u.hint.thisSize;
189  }
190 
191  if (ret && ptr)
192  XMEMCPY(ret, ptr, sz);
193 
194  if (ret) {
195  #ifdef WOLFSSL_DEBUG_MEMORY
196  TrackFree(ptr, func, line);
197  #else
198  TrackFree(ptr);
199  #endif
200  }
201 
202  return ret;
203  }
