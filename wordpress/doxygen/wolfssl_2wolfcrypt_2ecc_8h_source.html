191 /* Note on ALT_ECC_SIZE:
192  * The fast math code uses an array of a fixed size to store the big integers.
193  * By default, the array is big enough for RSA keys. There is a size,
194  * FP_MAX_BITS which can be used to make the array smaller when one wants ECC
195  * but not RSA. Some people want fast math sized for both RSA and ECC, where
196  * ECC won't use as much as RSA. The flag ALT_ECC_SIZE switches in an alternate
197  * ecc_point structure that uses an alternate fp_int that has a shorter array
198  * of fp_digits.
199  *
200  * Now, without ALT_ECC_SIZE, the ecc_point has three single item arrays of
201  * mp_ints for the components of the point. With ALT_ECC_SIZE, the components
202  * of the point are pointers that are set to each of a three item array of
203  * alt_fp_ints. While an mp_int will have 4096 bits of digit inside the
204  * structure, the alt_fp_int will only have 528 bits. A size value was added
205  * in the ALT case, as well, and is set by mp_init() and alt_fp_init(). The
206  * functions fp_zero() and fp_copy() use the size parameter. An int needs to
207  * be initialized before using it instead of just fp_zeroing it, the init will
208  * call zero. FP_MAX_BITS_ECC defaults to 528, but can be set to change the
209  * number of bits used in the alternate FP_INT.
210  *
211  * Do not enable ALT_ECC_SIZE and disable fast math in the configuration.
212  */
213 
214 #ifndef USE_FAST_MATH
215  #error USE_FAST_MATH must be defined to use ALT_ECC_SIZE
216 #endif

218 /* determine max bits required for ECC math */
219 #ifndef FP_MAX_BITS_ECC
220  /* check alignment */
221  #if ((MAX_ECC_BITS * 2) % DIGIT_BIT) == 0
222  /* max bits is double */
223  #define FP_MAX_BITS_ECC (MAX_ECC_BITS * 2)
224  #else
225  /* max bits is doubled, plus one digit of fudge */
226  #define FP_MAX_BITS_ECC ((MAX_ECC_BITS * 2) + DIGIT_BIT)
227  #endif
228 #else
229  /* verify alignment */
230  #if FP_MAX_BITS_ECC % CHAR_BIT
231  #error FP_MAX_BITS_ECC must be a multiple of CHAR_BIT
232  #endif
233 #endif
234 
235 /* determine buffer size */
236 #define FP_SIZE_ECC (FP_MAX_BITS_ECC/DIGIT_BIT)
