365  /* declare/free variable handling for async */
366  #ifdef WOLFSSL_ASYNC_CRYPT
367  #define DECLARE_VAR(VAR_NAME, VAR_TYPE, VAR_SIZE, HEAP) \
368  VAR_TYPE* VAR_NAME = (VAR_TYPE*)XMALLOC(sizeof(VAR_TYPE) * VAR_SIZE, HEAP, DYNAMIC_TYPE_WOLF_BIGINT);
369  #define DECLARE_VAR_INIT(VAR_NAME, VAR_TYPE, VAR_SIZE, INIT_VALUE, HEAP) \
370  VAR_TYPE* VAR_NAME = ({ \
371  VAR_TYPE* ptr = XMALLOC(sizeof(VAR_TYPE) * VAR_SIZE, HEAP, DYNAMIC_TYPE_WOLF_BIGINT); \
372  if (ptr && INIT_VALUE) { \
373  XMEMCPY(ptr, INIT_VALUE, sizeof(VAR_TYPE) * VAR_SIZE); \
374  } \
375  ptr; \
376  })
377  #define DECLARE_ARRAY(VAR_NAME, VAR_TYPE, VAR_ITEMS, VAR_SIZE, HEAP) \
378  VAR_TYPE* VAR_NAME[VAR_ITEMS]; \
379  int idx##VAR_NAME; \
380  for (idx##VAR_NAME=0; idx##VAR_NAME<VAR_ITEMS; idx##VAR_NAME++) { \
381  VAR_NAME[idx##VAR_NAME] = (VAR_TYPE*)XMALLOC(VAR_SIZE, HEAP, DYNAMIC_TYPE_WOLF_BIGINT); \
382  }
383  #define FREE_VAR(VAR_NAME, HEAP) \
384  XFREE(VAR_NAME, HEAP, DYNAMIC_TYPE_WOLF_BIGINT);
385  #define FREE_ARRAY(VAR_NAME, VAR_ITEMS, HEAP) \
386  for (idx##VAR_NAME=0; idx##VAR_NAME<VAR_ITEMS; idx##VAR_NAME++) { \
387  XFREE(VAR_NAME[idx##VAR_NAME], HEAP, DYNAMIC_TYPE_WOLF_BIGINT); \
388  }
389  #else
390  #define DECLARE_VAR(VAR_NAME, VAR_TYPE, VAR_SIZE, HEAP) \
391  VAR_TYPE VAR_NAME[VAR_SIZE]
392  #define DECLARE_VAR_INIT(VAR_NAME, VAR_TYPE, VAR_SIZE, INIT_VALUE, HEAP) \
393  VAR_TYPE* VAR_NAME = (VAR_TYPE*)INIT_VALUE
394  #define DECLARE_ARRAY(VAR_NAME, VAR_TYPE, VAR_ITEMS, VAR_SIZE, HEAP) \
395  VAR_TYPE VAR_NAME[VAR_ITEMS][VAR_SIZE]
